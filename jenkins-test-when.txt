pipeline {
    agent any

    stages {
   
        stage('dev') {
         when {
           allOf{
               branch 'develop'
               }
              }
            steps {
                echo 'executing dev branch'
            }
        }
         stage('master') {
         when {
          allOf{
               branch 'master'
               }
              }
            steps {
                echo 'executing master branch'
            }
        }
      
    }
}
